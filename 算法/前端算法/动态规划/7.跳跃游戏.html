    7.跳跃游戏

    zsy 2021/12/12

    # 方法一：使用动态规划

    /**
 * @param {number[]} nums
 * @return {boolean}
 */
var canJump = function (nums) {
    let n = nums.length
    let f = []
    f[0] = true  // 初始化条件
    for (let i = 1; i < n; i++) {
        f[i] = false
        for (let j = 0; j < i; j++) {
            if (f[j] && nums[j] >= i - j) {  // 子问题
                // 想要跳到位置j，必须要能跳到位置i
                // 想要跳到位置j，则需要当前能跳到的步数要大于两个的距离
                f[i] = true
                break
            }
        }
    }
    // console.log(f)  // [ true, true, true, true, true ]
    return f[n - 1]
};

let nums = [2, 3, 1, 1, 4]
let result = canJump(nums)
console.log(result)

    # 方法二：使用贪心算法

    • 假如当前在索引为0的位置，那么就标记一下可以跳到的最远位置
    • 遍历到索引为1的位置，那么也跳最远，然后更新最远的位置
    /**
 * @param {number[]} nums
 * @return {boolean}
 */
var canJump = function (nums) {
    let reach = 0
    let n = nums.length
    for (let i = 0; i < nums.length; i++) {
        if (i > reach) {  // 表示当前遍历的位置，是不能抵达的位置了
            return false
        }
        reach = Math.max(i + nums[i], reach)
    }
    return true
};
